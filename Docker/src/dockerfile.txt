FROM <имя>:<тег> - базовый образ с которого сделать наш образ

RUN - выполнить команду

WORKDIR - директория, в которой начинается работа контейнера

COPY <место на локальной машине откуда копировать> <место в образе куда копировать> - копирование файлов

CMD ["", ""] - выполнить при старте контейнера

ENV <имя переменной> <значение переменной> - задать переменную окружения

EXPOSE <номер порта> - информация о том какие порты могут быть проброшены (проброс не происходит)

ENTRYPOINT ["", ""]
